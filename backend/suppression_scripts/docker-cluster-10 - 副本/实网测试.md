## 🚀 部署流程

### 阶段1: VPS批量准备

#### 1.1 批量购买VPS
```bash
# 使用Vultr API批量创建（示例）
curl "https://api.vultr.com/v2/instances" \
  -X POST \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "region": "ewr",
    "plan": "vc2-1c-1gb",
    "os_id": 387,
    "label": "sybil-attack-node-1",
    "tag": "dht-attack"
  }'

# 重复20次，创建20台VPS
```

#### 1.2 批量配置SSH密钥
```bash
# 生成SSH密钥对
ssh-keygen -t ed25519 -f ~/.ssh/dht_attack_key -N ""

# 将公钥添加到所有VPS
for ip in $(cat vps_list.txt); do
  ssh-copy-id -i ~/.ssh/dht_attack_key.pub root@$ip
done
```

#### 1.3 批量初始化VPS
```bash
# 创建初始化脚本
cat > init_vps.sh << 'EOF'
#\!/bin/bash
# VPS初始化脚本

# 更新系统
apt-get update && apt-get upgrade -y

# 安装必要软件
apt-get install -y python3 python3-pip git curl wget

# 安装Python依赖
pip3 install requests flask

# 配置防火墙
ufw allow 22/tcp
ufw allow 8000:9000/tcp
ufw --force enable

# 优化系统参数
cat >> /etc/sysctl.conf << 'SYSCTL'
net.core.somaxconn = 1024
net.ipv4.tcp_max_syn_backlog = 2048
net.ipv4.ip_local_port_range = 10000 65000
SYSCTL
sysctl -p

echo "VPS初始化完成"
EOF

# 批量执行初始化
for ip in $(cat vps_list.txt); do
  echo "正在初始化 $ip..."
  scp init_vps.sh root@$ip:/tmp/
  ssh root@$ip "bash /tmp/init_vps.sh"
done
```

#### 关键性能指标 (KPI)
```python
# 监控脚本
import time
import requests

API_URL = "http://localhost:8080"
TASK_ID = "distributed-sybil_1715421234"

while True:
    # 获取任务状态
    response = requests.get(
        f"{API_URL}/api/sybil-attack/distributed/{TASK_ID}/status"
    )
    
    data = response.json()
    vps_assignments = data.get('vps_assignments', [])
    
    # 统计
    total_vps = len(vps_assignments)
    online_vps = sum(1 for v in vps_assignments if v['status'] == 'running')
    total_nodes = sum(v['nodes_count'] for v in vps_assignments)
    
    print(f"[{time.strftime('%H:%M:%S')}] VPS: {online_vps}/{total_vps}, 节点: {total_nodes}")
    
    # 检查异常
    if online_vps < total_vps * 0.9:
        print("⚠️ 告警: 超过10%的VPS离线！")
    
    time.sleep(5)
```
